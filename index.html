import dbconfig,psycopg2,wordcloud

# Connection
conn = psycopg2.connect(database=dbconfig.database,
                        host=dbconfig.host,
                        port=dbconfig.port,
                        user=dbconfig.user,
                        password=dbconfig.password)

# Cursor
c = conn.cursor()

Get all reviews for Laphroaig 10¶

In [13]:
c.execute("select review_text from review where review_id in (select review_id from metadata where topic='laphroaig 10');")

text = ' '.join((m[0] for m in c))

Create and display a word-cloud¶

In [14]:
import matplotlib.pyplot as plt 

wc = wordcloud.WordCloud(stopwords=set(['laphroaig', '10','color','nose','taste','finish']),
                       max_words=1000).generate(text)

default_colors = wc.to_array()
plt.title("Default colors")
plt.imshow(default_colors)
plt.axis("off")
plt.show()

No surprises there! ;)
